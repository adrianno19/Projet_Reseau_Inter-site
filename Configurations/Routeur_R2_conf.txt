!
! Configuration de la liaison vers R1
!
R2(config)# interface gigabitethernet 0/0
R2(config-if)# ip address 10.10.10.2 255.255.255.252
R2(config-if)# no shutdown
!
! Configuration de la liaison vers le switch L3
!
R2(config)# interface gigabitethernet 0/1
R2(config-if)# ip address 10.10.20.1 255.255.255.252
R2(config-if)# no shutdown
!
! Routage statique vers les réseaux du site 1 (via R1)
!
R2(config)# ip route 192.168.10.0 255.255.255.0 10.10.10.1
R2(config)# ip route 192.168.20.0 255.255.255.0 10.10.10.1
R2(config)# ip route 192.168.30.0 255.255.255.0 10.10.10.1
!
! Routage statique vers les réseaux du site 2 (via le switch L3)
!
R2(config)# ip route 172.16.10.0 255.255.255.0 10.10.20.2
R2(config)# ip route 172.16.20.0 255.255.255.0 10.10.20.2
R2(config)# ip route 172.16.30.0 255.255.255.0 10.10.20.2
